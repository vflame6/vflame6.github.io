{%- comment -%} Renders a language toggle when the current post has a `ref` shared by at least one other post. The `page.ref` guard prevents matching all the other posts that have no `ref` set at all (Liquid's `where: "ref", nil` would otherwise return every untagged post and produce a false toggle). {%- endcomment -%} {%- if page.ref -%} {%- assign all_docs = site.posts | concat: site.translations -%} {%- assign translations = all_docs | where: "ref", page.ref | sort: "lang" -%} {%- if translations.size > 1 -%}
{%- for t in translations -%} {%- if t.lang == "en" -%} {%- assign label = "🇬🇧 English" -%} {%- elsif t.lang == "ru" -%} {%- assign label = "🇷🇺 Русский" -%} {%- else -%} {%- assign label = t.lang -%} {%- endif -%} {%- if t.url == page.url -%} {{ label }} {%- else -%} {{ label }} {%- endif -%} {%- unless forloop.last %} · {% endunless -%} {%- endfor -%}
{%- endif -%} {%- endif -%}
